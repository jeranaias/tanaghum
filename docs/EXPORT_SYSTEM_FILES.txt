Integration:
------------
Only 2 steps required:

1. Add to generator.html <head>:
   <link rel="stylesheet" href="css/export-dialog.css">

2. Update export button handler:
   import { exportDialog } from './js/ui/export-dialog.js';
   
   $('#export-btn').addEventListener('click', () => {
     const lesson = StateManager.get('lesson');
     exportDialog.show(lesson);
   });
